\begin{itemize}
\tightlist
\item $2 \leq n \leq 1000$
\item $n-1 \leq m \leq 5 \times 10 ^ 5$

\item $1 \leq p_i \leq 10 ^ 9$ for $i = 1, 2, \ldots, k$

\item $1 \leq w_i \leq 10^9$ for $i = 1, 2, \ldots, m$
\item $1 \leq c_i \leq k$ for $i = 1, 2, \ldots, m$
